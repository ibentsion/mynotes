Create `docker/` directory with two Dockerfiles.

**docker/Dockerfile.gpu** — for the remote WSL2 GPU agent:

```dockerfile
FROM nvidia/cuda:12.8.0-cudnn-runtime-ubuntu22.04

ENV DEBIAN_FRONTEND=noninteractive

# System deps: Python 3.13 via deadsnakes PPA + OpenCV + pdf2image system libs
RUN apt-get update && apt-get install -y --no-install-recommends \
    software-properties-common \
    && add-apt-repository ppa:deadsnakes/ppa \
    && apt-get update && apt-get install -y --no-install-recommends \
    python3.13 \
    python3.13-dev \
    python3.13-venv \
    python3-pip \
    git \
    curl \
    poppler-utils \
    libpoppler-cpp-dev \
    libgl1 \
    libglib2.0-0 \
    && rm -rf /var/lib/apt/lists/*

# Make python3.13 the default python3/python
RUN update-alternatives --install /usr/bin/python3 python3 /usr/bin/python3.13 1 \
    && update-alternatives --install /usr/bin/python python python3.13 1

# Install pip for 3.13
RUN curl -sS https://bootstrap.pypa.io/get-pip.py | python3.13

# Pre-install all project deps (GPU torch from cu128 index)
COPY requirements.txt /tmp/requirements.txt
RUN pip install --no-cache-dir -r /tmp/requirements.txt

# Install clearml-agent so the container can act as an agent executor
RUN pip install --no-cache-dir clearml-agent

WORKDIR /root
```

**docker/Dockerfile.cpu** — for local development parity testing:

```dockerfile
FROM python:3.13-slim-bookworm

ENV DEBIAN_FRONTEND=noninteractive

RUN apt-get update && apt-get install -y --no-install-recommends \
    git \
    curl \
    poppler-utils \
    libpoppler-cpp-dev \
    libgl1 \
    libglib2.0-0 \
    && rm -rf /var/lib/apt/lists/*

# Install CPU torch deps — same package versions but CPU wheel index
RUN pip install --no-cache-dir \
    pdf2image==1.17.0 \
    opencv-python-headless==4.13.0.92 \
    numpy==2.4.4 \
    pandas==3.0.2 \
    clearml==2.1.5 \
    streamlit==1.56.0 \
    "scikit-learn>=1.5" \
    openai>=1.0 \
    optuna==4.8.0

# Install CPU torch separately from the CPU wheel index
RUN pip install --no-cache-dir \
    --extra-index-url https://download.pytorch.org/whl/cpu \
    torch==2.11.0+cpu

WORKDIR /workspace
```

```yaml

  mynotes-cpu:
    build:
      context: ..
      dockerfile: docker/Dockerfile.cpu
    image: mynotes-cpu
    volumes:
      - ..:/workspace
    working_dir: /workspace
    command: python -c "import torch, cv2, clearml; print('all imports ok')"
    environment:
      - CLEARML_API_ACCESS_KEY=${CLEARML_API_ACCESS_KEY:-}
      - CLEARML_API_SECRET_KEY=${CLEARML_API_SECRET_KEY:-}
      - CLEARML_API_HOST=${CLEARML_API_HOST:-}
```

**scripts/start_clearml_agent_docker.sh** — builds the GPU image and starts the agent in docker mode. Must be run on the WSL2 GPU machine:

```bash
#!/usr/bin/env bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"

IMAGE_NAME="mynotes-gpu"
QUEUE="${1:-ofek}"

echo "Building $IMAGE_NAME from $REPO_ROOT/docker/Dockerfile.gpu..."
docker build -t "$IMAGE_NAME" -f "$REPO_ROOT/docker/Dockerfile.gpu" "$REPO_ROOT"

echo "Starting clearml-agent in docker mode on queue: $QUEUE"
echo "Image: $IMAGE_NAME"
echo "Press Ctrl+C to stop."

clearml-agent daemon \
    --queue "$QUEUE" \
    --docker "$IMAGE_NAME" \
    --docker-args "--gpus all --shm-size=8g" \
    --foreground
```

Make the script executable: `chmod +x scripts/start_clearml_agent_docker.sh`

The script accepts an optional queue name as $1, defaulting to "ofek". This matches the existing ClearML queue used by the project.
